**2a. Append new `publish:` job at end of `jobs:` section (after `lint-workflow:` block ending line 1670)**

Add the following job with these EXACT properties (every bullet is load-bearing for a requirement):

```yaml
  # ── Phase 4 publish job ──
  # Runs on a green smoke-test only. Downloads the `cuda-wheel` artifact from
  # the build job (byte-identical to what Phase 2 produced — no repackaging),
  # computes release metadata, assembles the forensics body, and uploads the
  # wheel as an asset on a GitHub Release at tag `v<base_version>-cu<maj><min>-win`.
  # The gate is STRUCTURAL: `needs: [build, smoke-test]` + a lint-workflow
  # grep-assert below. Removing the gate is a grep-detectable regression.
  # See upstream #1543 for why smoke-test is a hard publish gate.
  publish:
    needs: [build, smoke-test]                          # PUB — hard gate, structural.
    runs-on: ubuntu-latest                              # PUB-01 — cheap minutes; no CUDA needed for asset upload.
    timeout-minutes: 10                                 # Sanity bound; typical runtime &lt; 2 min.
    permissions:
      contents: write                                   # Minimum scope required by softprops/action-gh-release@v2.
    defaults:
      run:
        shell: bash
    steps:
      - name: 'Publish: Download wheel artifact'
        uses: actions/download-artifact@v4              # PUB — v4 is the mandatory pair for Phase 2's upload-artifact@v4.
        timeout-minutes: 5
        with:
          name: cuda-wheel                              # SINGULAR — Phase 2 contract (see RESEARCH.md Pitfall 4).
          path: ./dist

      - name: 'Publish: Compute release metadata'
        id: meta
        timeout-minutes: 2
        run: |
          set -euo pipefail
          WHEEL=$(ls dist/*.whl | head -1)
          if [[ -z "$WHEEL" ]]; then
            echo "::error::No wheel found in dist/ after artifact download (expected `name: cuda-wheel` from build job)"
            exit 1
          fi
          WHEEL_NAME=$(basename "$WHEEL")
          WHEEL_SIZE=$(stat -c%s "$WHEEL")
          WHEEL_SIZE_MIB=$(awk -v s="$WHEEL_SIZE" 'BEGIN { printf "%.1f", s/1024/1024 }')
          WHEEL_SHA256=$(sha256sum "$WHEEL" | cut -d' ' -f1)
          BUILD_DATE=$(date -u +%Y-%m-%d)
          # Parse base_version from `llama_cpp_python-<base>+cu126.ll<sha>-cp...` (Phase 2 locked scheme).
          BASE_VER=$(echo "$WHEEL_NAME" | sed -E 's/^llama_cpp_python-([0-9]+\.[0-9]+\.[0-9]+)\+.*/\1/')
          # Derive cu-tag from inputs.cuda_version (NOT hardcoded — see RESEARCH.md Pitfall 8).
          CU_TAG=$(echo '${{ inputs.cuda_version }}' | awk -F. '{printf "cu%s%s", $1, $2}')
          TAG="v${BASE_VER}-${CU_TAG}-win"
          # Sanity assert: tag must match the canonical shape (belt-and-suspenders vs Phase 2 filename drift).
          if [[ ! "$TAG" =~ ^v[0-9]+\.[0-9]+\.[0-9]+-cu[0-9]+-win$ ]]; then
            echo "::error::Parsed tag '$TAG' does not match ^v\\d+\\.\\d+\\.\\d+-cu\\d+-win\$ — wheel filename scheme may have drifted from Phase 2 contract"
            exit 1
          fi
          {
            echo "wheel_name=$WHEEL_NAME"
            echo "wheel_path=$WHEEL"
            echo "wheel_size_bytes=$WHEEL_SIZE"
            echo "wheel_size_mib=$WHEEL_SIZE_MIB"
            echo "wheel_sha256=$WHEEL_SHA256"
            echo "build_date=$BUILD_DATE"
            echo "tag=$TAG"
            echo "base_version=$BASE_VER"
            echo "cu_tag=$CU_TAG"
          } >> "$GITHUB_OUTPUT"

      - name: 'Publish: Assemble release body'
        id: body
        timeout-minutes: 2
        run: |
          set -euo pipefail
          # Random delimiter prevents collision with `EOF` literals in body content (see RESEARCH.md Pitfall 3).
          DELIM="RELEASE_BODY_$(date +%s)_${RANDOM}"
          {
            echo "content&lt;&lt;${DELIM}"
            cat &lt;&lt;BODY
          Windows CUDA ${{ inputs.cuda_version }} wheel, built ${{ steps.meta.outputs.build_date }} against llama.cpp @ \`${{ needs.build.outputs.llama_sha }}\`.

          ## Install

          1. Verify your NVIDIA driver is &ge; 561.17 (\`nvidia-smi\`).
          2. Download \`${{ steps.meta.outputs.wheel_name }}\` below.
          3. \`pip install path/to/${{ steps.meta.outputs.wheel_name }}\`.
          4. Verify: \`python -c "from llama_cpp import Llama; print('ok')"\`.

          ## Build forensics

          | Property | Value |
          |---|---|
          | llama.cpp SHA | \`${{ needs.build.outputs.llama_sha }}\` |
          | Base version | \`${{ steps.meta.outputs.base_version }}\` |
          | CUDA version | \`${{ inputs.cuda_version }}\` |
          | MSVC toolset | \`${{ needs.build.outputs.selected_toolset }}\` |
          | Python | \`${{ inputs.python_version }}\` |
          | Wheel size | \`${{ steps.meta.outputs.wheel_size_bytes }}\` bytes (${{ steps.meta.outputs.wheel_size_mib }} MiB) |
          | Wheel SHA-256 | \`${{ steps.meta.outputs.wheel_sha256 }}\` |
          | Build date (UTC) | \`${{ steps.meta.outputs.build_date }}\` |
          | Dispatch run | https://github.com/${{ github.repository }}/actions/runs/${{ github.run_id }} |
          | Smoke-test | green (publish gated on \`needs: [build, smoke-test]\`) |
          BODY
            echo "${DELIM}"
          } >> "$GITHUB_OUTPUT"

      - name: 'Publish: Upload wheel as GitHub Release asset'
        uses: softprops/action-gh-release@v2            # PUB-02 — locked.
        timeout-minutes: 5
        with:
          tag_name: ${{ steps.meta.outputs.tag }}
          name: ${{ steps.meta.outputs.tag }}
          body: ${{ steps.body.outputs.content }}
          files: dist/*.whl
          target_commitish: ${{ github.sha }}           # Pin tag to exact build commit (reproducibility; RESEARCH.md Pitfall 1).
          fail_on_unmatched_files: true                 # Fail-loud if dist/ is empty (vs silent 0-exit).
          prerelease: false
          make_latest: 'true'                           # String per action.yml; marks this as the "Latest" release badge.
          # overwrite_files defaults to true — CONTEXT.md clobber semantics for same-filename re-dispatch.

      - name: 'Publish: Green forensics summary'
```
