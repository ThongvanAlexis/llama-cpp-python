```bash
# Run metadata
gh run view "$RUN_ID" --json conclusion,status,displayTitle,createdAt,updatedAt,jobs \
  -q '{conclusion, status, duration_min: ((.updatedAt | fromdateiso8601) - (.createdAt | fromdateiso8601)) / 60, jobs: [.jobs[] | {name, conclusion, started: .startedAt, completed: .completedAt}]}' \
  > .planning/phases/04-publish-consumer-ux/04-02-run-meta.json

# Publish job step-by-step log (forensics)
gh run view "$RUN_ID" --log --job=$(gh run view "$RUN_ID" --json jobs -q '.jobs[] | select(.name=="publish") | .databaseId') \
  > .planning/phases/04-publish-consumer-ux/04-02-publish-log.txt

# Derive expected tag
BASE_VER=$(grep -E '^__version__' llama_cpp/__init__.py | sed -E 's/.*"([0-9.]+)".*/\1/')
TAG="v${BASE_VER}-cu126-win"
echo "Expected tag: $TAG"

# Release view
gh release view "$TAG" --json tagName,name,body,assets,targetCommitish,createdAt,isLatest \
  > .planning/phases/04-publish-consumer-ux/04-02-release.json

# Asset SHA-256 from the uploaded wheel (download + hash, then compare to what the publish job's 'meta' step computed)
ASSET_NAME=$(jq -r '.assets[0].name' .planning/phases/04-publish-consumer-ux/04-02-release.json)
gh release download "$TAG" --pattern "$ASSET_NAME" --dir /tmp/release-verify
UPLOADED_SHA=$(sha256sum /tmp/release-verify/"$ASSET_NAME" | cut -d' ' -f1)
STEP_SHA=$(grep -oE 'wheel_sha256=[a-f0-9]+' .planning/phases/04-publish-consumer-ux/04-02-publish-log.txt | head -1 | cut -d= -f2)
echo "Uploaded asset SHA-256: $UPLOADED_SHA"
echo "Publish job reported  : $STEP_SHA"
[[ "$UPLOADED_SHA" == "$STEP_SHA" ]] && echo "MATCH (byte-identical)" || echo "MISMATCH (investigate)"
```
